Differentiation Formulas

ðŸ“œ Limit:

\[ \text{ (i) } \lim_{x \to a} \frac{f(x)}{f(x)} = \frac{\lim_{x \to a} f(x)}{\lim_{x \to a} f(x)} \]
\[ \text{ (ii) } \lim_{x \to a} (c) = c \text{ [c constant]} \]
\[ \text{ (iii) } \lim_{x \to a} (cx) = c \lim_{ x \to a} (x) \text{ [c constant]} \]
\[ \text{ (iv) } \lim_{x \to 0} \frac{ \sin x} {x} = 1 \]
\[ \text{ (v) } \lim_{x \to 0} \frac{x} {\sin x} = 1 \]
\[ \text{ (vi) } \lim_{x \to 0} \frac{ \tan x} {x} = 1 \]
\[ \text{ (vii) } \lim_{x \to 0} \frac{x} { \tan x} = 1 \]
\[ \text{ (viii) } \lim_{x \to 0} \cos x = 1 \]
\[ \text{ (ix) } \lim_{x \to 0} \frac{1} {\cos x} = 1 \]
\[ \text{ (x) } \lim_{x \to a} \frac{x^n - a^n} {x -a} = na^{n-1} \]
\[ \text{ (xi) } \lim_{x \to a} \frac{x^m - a^m} {x^n -a^n} = \frac{m}{n} \left(a^{m-n }\right) \]
\[ \text{ (xii) } \lim_{x \to 0} \frac{\ln (1 + x) }{x} = 1 \]
\[ \text{ (xiii) } \lim_{x \to \infty} \left( 1+ \frac{m}{x} \right)^x = e^m \]
\[ \text{ (xiv) } \lim_{x \to 0} \left( 1+ x \right)^{ \frac{1}{x}} = e \]
\[ \text{ (xv) } \lim_{x \to \infty } \left( 1+ \frac{1}{x} \right)^x = e \]
\[ \text{ (xvi) } \lim_{x \to a} \frac{e^x-1}{x} = 1 \]
\[ \text{ (xvii) } \lim_{x \to 0} \frac{ \sin ^{-1} x}{x} = 1 \]
\[ \text{ (xviii) } \lim_{x \to 0} \frac{x}{\sin ^{-1}x} = 1 \]
\[ \text{ (xix) } \lim_{x \to 0} \sin x = \lim_{x \to 0} x \]
\[ \text{ (xx) } \lim_{x \to 0} \frac{ \tan ^{-1} x}{x} = 1 \]
\[ \text{ (xxi) } \lim_{x \to 0} \frac{x}{\tan ^{-1}x} = 1 \]
\[ \text{ (xxii) } \lim_{x \to h} \frac{f(x + h) - f(x)}{h} = \frac{d}{dx} f(x) \]
\[ \text{ (xxiii) } \lim_{x \to \infty} \left( 1+ \frac{m}{x} \right)^{nx} = e^{mn} \]
\[ \text{ (xxiv) } \lim_{x \to \infty} \left( 1+ {m}{x} \right)^\frac{n}{x} = e^{mn} \]

ðŸ“œ Differentiation:

\[ \text{ (i) } \frac{d}{dx}(x^n) = n x^{n-1} \]
\[ \text{ (ii) } \frac{d}{dx}(\sin(nx)) = n \cos(nx) \]
\[ \text{ (iii) } \frac{d}{dx}(\cos(x)) = -\sin(x) \]
\[ \text{ (iv) } \frac{d}{dx}(\tan(x)) = \sec^2(x) \]
\[ \text{ (v) } \frac{d}{dx}(\cot(x)) = -\csc^2(x) \]
\[ \text{ (vi) } \frac{d}{dx}(\sec(x)) = \sec(x) \tan(x) \]
\[ \text{ (vii) } \frac{d}{dx}(\csc(x)) = -\csc(x) \cot(x) \]
\[ \text{ (viii) } \frac{d}{dx}(e^x) = e^x \]
\[ \text{ (ix) } \frac{d}{dx}(\ln(x)) = \frac{1}{x} \]
\[ \text{ (x) } \frac{d}{dx}(a^x) = a^x \ln(a) \]
\[ \text{ (xi) } \frac{d}{dx}(\log_a(x)) = \frac{1}{x \ln(a)} \]
\[ \text{ (xii) } \frac{d}{dx}(u v) = u\frac{dv}{dx} + v\frac{du}{dx} \]
\[ \text{ (xiii) } \frac{d}{dx}\left(\frac{u}{v}\right) = \frac{v \frac{du}{dx} - u \frac{dv}{dx}}{v^2} \]
\[ \text{ (xiv) } \frac{dy}{dx} = \frac{dy}{dz}\cdot\frac{dz}{dx} \]
\[ \text{ (xv) } \frac{d}{dx}(\sin^{-1}(x)) = \frac{1}{\sqrt{1 - x^2}} \]
\[ \text{ (xvi) } \frac{d}{dx}(\cos^{-1}(x)) = -\frac{1}{\sqrt{1 - x^2}} \]
\[ \text{ (xvii) } \frac{d}{dx}(\tan^{-1}(x)) = \frac{1}{1 + x^2} \]
\[ \text{ (xviii) } \frac{d}{dx}(\cot^{-1}(x)) = -\frac{1}{1 + x^2} \]
\[ \text{ (xix) } \frac{d}{dx}(\sec^{-1}(x)) = \frac{1}{|x| \sqrt{x^2 - 1}} \]
\[ \text{ (xx) } \frac{d}{dx}(\csc^{-1}(x)) = -\frac{1}{|x| \sqrt{x^2 - 1}} \]
\[ \text{ (xxi) } f'(x) = \frac{dy}{dx} = \lim_{h \to 0} \frac{f(x+h) - f(x) }{ h } \]
\[ \text{ (xxii) } \frac{d}{dx}(u \pm v) = \frac{du}{dx} \pm \frac{dv}{dx} \]
\[ \text{ (xxiii) } \frac{d}{dx}(uvw) = \frac{\frac{d}{dx}(u)}{u} + \frac{\frac{d}{dx}(v)}{v} + \frac{\frac{d}{dx}(w)}{w} \]
\[ \text{ (xxiv) } \frac{d}{dx}(u^v) = u^v \frac{d}{dx} \left( v \ln(u) \right) \]

ðŸ“œ Nth Derivative:

\[ \text{(i) If } y = x^n, \text{ then } y_n = n! \]
\[ \text{(ii) If } y = e^{ax + b} , \text{ then } y_n = a^n e^{ax + b} \]
\[ \text{(iii) If } y = a^x, \text{ then } y_n = a^x(\ln a )^n \]
\[ \begin{aligned} \text{(iv) If } y &= \sin (ax + b), \\ \text{ then } y_n &= a^n \sin[\frac{n \pi}{2} + ax + b] \end{aligned}\]
\[ \begin{aligned} \text{(v) If } y &= \cos (ax + b), \\ \text{ then } y_n &= a^n \cos[\frac{n \pi}{2} + ax + b] \end{aligned} \]
\[ \text{(vi) If } y = \ln x, \text{ then } y_n = \frac{(-1)^{n-1} (n-1)! }{x^n} \]
\[ \text{(vi) If } y = \ln x, \text{ then } y_n = \frac{(-1)^{n-1} (n-1)! }{x^n} \]
\[ \begin{aligned} \text{(vi) If } y &= e^{ax} \sin \left(bx + c \right), \text{ then } \\ y_n &= \left(a^2 + b^2 \right)^{\frac{n}{2}} e^{ax} \sin \left(bx + c + n \tan^{-1} \frac{b}{a}\right) \end{aligned} \]
\[ \begin{align*} \text{(vi) If } y &= e^{ax} \cos \left(bx + c \right), \text{ then } \\ y_n &= \left(a^2 + b^2 \right)^{\frac{n}{2}} e^{ax} \cos \left(bx + c + n \tan^{-1} \frac{b}{a}\right) \end{align*} \]

ðŸ“œ Tangent & Perpendicular:

\[ \begin{aligned} \text{(i) The equation of tangent at }(x_1,y_1): \\ \\ y - y_1 = f'(x)(x-x_1) \end{aligned} \]
\[ \begin{aligned} \text{(i) The equation of perpendicular at }(x_1,y_1): \\ \\ x - x_1 + f'(x)(y - y_1) = 0 \end{aligned} \]
